# Hier stehen Kommentare
# Diese sollen nicht übernommen werden


# Man achte auf den abschließenden ':'

# Zuerst werden bestimmte Zeichen mit einem Character verbunden
# Hierbei wird das Endbild aus verschiedenen Einzelbildern zusammengesetzt
# Es ist kein abschließendes Komma nötig
C:Name:Pfad_zu_Bild1,Pfad_zu_Bild2,Pfad_zu_Bild3,...:Team:

# Nun kann man diesem Character über das Schlüsselwort 'Item' Gegenstände geben
# Wird an die Item-ID ein 'e' angehängt, so wird dieses ausgerüstet
# Die Item-Liste kann beliebig lang sein
Item:Name:Item1[e]:Item2[e]:...:

# Man kann auch ein Hintergrundbild festlegen
backgroundimg:Pfad_zum_Bild:


# Nun kann man die Karte erstellen
# Aus technischen Gründen muss sie quadratisch sein
# Als freies Feld können alle nicht definierten Zeichen genutzt werden
C___

___C
